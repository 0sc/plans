POST /users
Create a user

Create User

This endpoint is used to create a new user profile.

Supported Formats

json

Errors

422 Invalid parameter sent

Examples

// Does not create user with invalid email
POST /users
{
  "user": {
    "name": "Oscar",
    "email": "454@#(*)",
    "password": "tcpu1asg6zv"
  }
}
422
{
  "users": [
    "Email is invalid"
  ]
}
// Does not create user with empty email
POST /users
{
  "user": {
    "name": "Oscar",
    "email": "",
    "password": "fvbuxyd1sxx"
  }
}
422
{
  "users": [
    "Email can't be blank",
    "Email is invalid"
  ]
}
// Does not create user with short name
POST /users
{
  "user": {
    "name": "e",
    "email": "jarvis.schmeler@kerluke.com",
    "password": "2mhsaduqwa"
  }
}
422
{
  "users": [
    "Name is too short (minimum is 2 characters)"
  ]
}
// Does not create user with empty name
POST /users
{
  "user": {
    "name": "",
    "email": "email@email.com",
    "password": ""
  }
}
422
{
  "users": [
    "Password can't be blank",
    "Name can't be blank",
    "Name is too short (minimum is 2 characters)",
    "Email has already been taken"
  ]
}
// Does not create user with empty params
POST /users
{
  "user": {
    "name": "",
    "email": "",
    "password": ""
  }
}
422
{
  "users": [
    "Password can't be blank",
    "Name can't be blank",
    "Name is too short (minimum is 2 characters)",
    "Email can't be blank",
    "Email is invalid"
  ]
}
// Does not create user with long name
POST /users
{
  "user": {
    "name": "i9elfyek180bf1moqawolp1j6wpp7cq1dia5yvfxnum12qqdquu",
    "email": "bertram_kirlin@murphy.org",
    "password": "npxko5it9lc"
  }
}
422
{
  "users": [
    "Name is too long (maximum is 50 characters)"
  ]
}
// Does not create user with empty password
POST /users
{
  "user": {
    "name": "Oscar",
    "email": "antonietta.cain@quigley.io",
    "password": ""
  }
}
422
{
  "users": [
    "Password can't be blank"
  ]
}
// Creates user with valid params
POST /users
{
  "user": {
    "name": "Oscar",
    "email": "deon@johnson.io",
    "password": "nt5sqe611w"
  }
}
201
{
  "user": {
    "name": "Oscar",
    "email": "deon@johnson.io",
    "created_at": "2015-11-30T01:19:21.070Z",
    "updated_at": "2015-11-30T01:19:21.070Z"
  }
}
